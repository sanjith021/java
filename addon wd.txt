        Account ac = new Account("Sanju", 210000);
        System.out.println("Account Holder Name: " + ac.getAccholder());
        System.out.println("Holder Balance: " + ac.getBalance());

            ac.withdrawAmount(400000);
        ac.setBalance(10000);
        ac.setAccholder("Dhilip");
        System.out.println("Update Account Holder Name: " + ac.getAccholder());
        System.out.println("Update Holder Balance: " + ac.getBalance());
    }
}

class Account {
    private String Accholder;
    private int Balance;

    Account(String Accholder, int Balance) {
        this.Accholder = Accholder;
        this.Balance = Balance;
    }

    public String getAccholder() {
        return Accholder;
    }
    public int getBalance() {
        return Balance;
    }
    public void setAccholder(String Accholder) {
        this.Accholder = Accholder;
    }
    public void setBalance(int amount){
        if(Balance>0){
            Balance+=amount;
            System.out.println("deposit successfully total balance:"+Balance);
        }
    }
    public void creditAmount(int amount) {
        this.Balance += amount;
        System.out.println("Amount credited: " + amount);
        System.out.println("Update Balance: " + this.Balance);
    }
    public void withdrawAmount(int Amount){
        this.Balance -=Amount;
        System.out.println("withdraw:" + Amount);
        System.out.println("Update Balance: " + this.Balance);
    }
}
